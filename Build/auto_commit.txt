@echo off
set source_folder="C:\path\to\folder\a"
set destination_folder="C:\path\to\folder\b"
set exclude_files="%~nx0" "shortcut.lnk"

echo Copying files from %source_folder% to %destination_folder%...
xcopy /s /y /exclude:%exclude_files% %source_folder%\*.* %destination_folder%\

set /p commit_message="Enter commit message: "
echo Committing changes to Git...
cd %destination_folder%
git add .
git commit -m "%commit_message%"
git push origin master

echo Done.
